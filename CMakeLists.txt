cmake_minimum_required(VERSION 3.15)
project(xcdoc LANGUAGES CXX C)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(TESTS_RESOURCE_DIR ${CMAKE_CURRENT_SOURCE_DIR}/tests/res)
add_compile_definitions(XCDOC_TESTS_RESOURCE_DIR="${TESTS_RESOURCE_DIR}")
option(BUILD_TESTING "Build tests" ON)
find_program(RE2C re2c REQUIRED)
find_program(FLEX flex REQUIRED)
add_subdirectory(src)
if(BUILD_TESTING)
    enable_testing()
    add_subdirectory(tests)
endif()
